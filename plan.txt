### 3. API Calls
```dart
// lib/services/medicine_service.dart
class MedicineService {
  final ApiService _api;
  
  MedicineService(this._api);
  
  Future<List<Medicine>> getAllMedicines() async {
    final response = await http.get(
      Uri.parse('${ApiService.baseUrl}/medicines'),
      headers: _api.headers,
    );
    
    if (response.statusCode == 200) {
      List<dynamic> data = jsonDecode(response.body);
      return data.map((json) => Medicine.fromJson(json)).toList();
    } else {
      throw Exception('Failed to load medicines');
    }
  }
  
  Future<Medicine?> scanBarcode(String barcode) async {
    final response = await http.get(
      Uri.parse('${ApiService.baseUrl}/medicines/barcode/$barcode'),
      headers: _api.headers,
    );
    
    if (response.statusCode == 200) {
      return Medicine.fromJson(jsonDecode(response.body));
    }
    return null;
  }
}
```

```dart
class MedicineProvider extends ChangeNotifier {
```

```java
public class MedicineService {

            .collect(Collectors.toList());
    }
    
    // ENHANCED EXPIRY MANAGEMENT
    public List<MedicineResponse> getExpiringSoon(Integer days) {
        LocalDate today = LocalDate.now();
        LocalDate futureDate = today.plusDays(days);
        log.debug("Fetching medicines expiring between {} and {}", today, futureDate);
        return medicineRepository.findExpiringBetween(today, futureDate)
            .stream()
            .map(this::toResponse)
            .collect(Collectors.toList());
    }
    
    public List<MedicineResponse> getExpiredMedicines() {
        LocalDate today = LocalDate.now();
        return medicineRepository.findExpiredMedicines(today)

            .collect(Collectors.toList());

        medicine.setIsActive(false);
        medicineRepository.save(medicine);
        log.info("Deactivated medicine: {}", medicine.getName());
    }

    public void reactivateMedicine(Long id) {
        Medicine medicine = medicineRepository.findById(id)
            .orElseThrow(() -> new ResourceNotFoundException("Medicine not found"));
        medicine.setIsActive(true);
        medicineRepository.save(medicine);
        log.info("Reactivated medicine: {}", medicine.getName());
    }
    
    // ENHANCED VALIDATION
    private void validateMedicine(Medicine medicine) {
        // Existing validations...
        
        // NEW VALIDATIONS
        if (medicine.getBarcode() != null && !medicine.getBarcode().isEmpty()) {
            if (!medicine.getBarcode().matches("^[0-9]{8,13}$")) {
                throw new IllegalArgumentException("Invalid barcode format");
            }
        }
        
        if (medicine.getMinStockLevel() != null && medicine.getMaxStockLevel() != null) {
            if (medicine.getMinStockLevel() >= medicine.getMaxStockLevel()) {
                throw new IllegalArgumentException("Min stock level must be less than max stock level");
            }
        }
        
        if (medicine.getCostPrice() != null && medicine.getSellingPrice() != null) {
            if (medicine.getCostPrice() > medicine.getSellingPrice()) {
                log.warn("Cost price is higher than selling price for: {}", medicine.getName());
            }
        }
        
        if (medicine.getManufactureDate() != null && medicine.getExpiryDate() != null) {
            if (medicine.getManufactureDate().isAfter(medicine.getExpiryDate())) {
                throw new IllegalArgumentException("Manufacture date cannot be after expiry date");
            }
        }
    }
    
    // ENHANCED RESPONSE MAPPING
    private MedicineResponse toResponse(Medicine medicine) {
        LocalDate today = LocalDate.now();
        Integer daysUntilExpiry = null;
        Boolean isExpired = false;
        Boolean isExpiringSoon = false;
        
        if (medicine.getExpiryDate() != null) {
            daysUntilExpiry = (int) ChronoUnit.DAYS.between(today, medicine.getExpiryDate());
            isExpired = medicine.getExpiryDate().isBefore(today);
            isExpiringSoon = daysUntilExpiry <= 30 && daysUntilExpiry > 0;
        }
        
        Boolean isLowStock = medicine.getStockQuantity() < medicine.getMinStockLevel();
        
        Double profitMargin = null;
        if (medicine.getCostPrice() != null && medicine.getSellingPrice() != null && medicine.getCostPrice() > 0) {
            profitMargin = ((medicine.getSellingPrice() - medicine.getCostPrice()) / medicine.getCostPrice()) * 100;
        }
        
        return MedicineResponse.builder()
            .id(medicine.getId())
            .name(medicine.getName())
            .category(medicine.getCategory())
            .sellingPrice(medicine.getSellingPrice())
            .costPrice(medicine.getCostPrice())
            .stockQuantity(medicine.getStockQuantity())
            .expiryDate(medicine.getExpiryDate())
            .manufacturer(medicine.getManufacturer())
            .barcode(medicine.getBarcode())
            .sku(medicine.getSku())
            .batchNumber(medicine.getBatchNumber())
            .manufactureDate(medicine.getManufactureDate())
            .dosageForm(medicine.getDosageForm())
            .strength(medicine.getStrength())
            .genericName(medicine.getGenericName())
            .brandName(medicine.getBrandName())
            .prescriptionRequired(medicine.getPrescriptionRequired())
            .minStockLevel(medicine.getMinStockLevel())
            .maxStockLevel(medicine.getMaxStockLevel())
            .unitOfMeasure(medicine.getUnitOfMeasure())
            .rackLocation(medicine.getRackLocation())
            .taxRate(medicine.getTaxRate())
            .imageUrl(medicine.getImageUrl())
            .activeIngredient(medicine.getActiveIngredient())
            .storageConditions(medicine.getStorageConditions())
            .isActive(medicine.getIsActive())
            .isLowStock(isLowStock)
            .isExpiringSoon(isExpiringSoon)
            .isExpired(isExpired)
            .daysUntilExpiry(daysUntilExpiry)
            .profitMargin(profitMargin)
            .build();
    }
}
```

```java
@Service
@Slf4j
public class ImageUploadService {
    
    @Value("${file.upload.dir:uploads/medicines}")
    private String uploadDir;
    
    public String uploadImage(MultipartFile file, Long medicineId) throws IOException {
        // Validate file
        if (file.isEmpty()) {
            throw new IllegalArgumentException("File is empty");
        }
        
        String contentType = file.getContentType();
        if (!contentType.startsWith("image/")) {
            throw new IllegalArgumentException("Only image files are allowed");
        }
        
        // Create upload directory if not exists
        Path uploadPath = Paths.get(uploadDir);
        if (!Files.exists(uploadPath)) {
            Files.createDirectories(uploadPath);
        }
        
        // Generate unique filename
        String originalFilename = file.getOriginalFilename();
        String extension = originalFilename.substring(originalFilename.lastIndexOf("."));
        String filename = "medicine_" + medicineId + "_" + System.currentTimeMillis() + extension;
        
        // Save file
        Path filePath = uploadPath.resolve(filename);
        Files.copy(file.getInputStream(), filePath, StandardCopyOption.REPLACE_EXISTING);
        
        log.info("Uploaded image: {}", filename);
        return "/uploads/medicines/" + filename;
    }
    
    public void deleteImage(String imageUrl) {
        try {
            Path filePath = Paths.get(uploadDir, imageUrl.substring(imageUrl.lastIndexOf("/") + 1));
            Files.deleteIfExists(filePath);
            log.info("Deleted image: {}", imageUrl);
        } catch (IOException e) {
            log.error("Failed to delete image: {}", imageUrl, e);
        }
    }
}
```

```java
            Long medicineId, 
            StockAdjustmentRequest request, 
            String username) {
        
        Medicine medicine = medicineRepository.findById(medicineId)
            .orElseThrow(() -> new ResourceNotFoundException("Medicine not found"));
        
        int previousQty = medicine.getStockQuantity();
        int adjustmentAmount = request.getQuantity();
        
        // For SALE, DAMAGE, EXPIRY, THEFT, TRANSFER_OUT - reduce stock
        if (request.getType() == AdjustmentType.SALE ||
            request.getType() == AdjustmentType.DAMAGE ||
            request.getType() == AdjustmentType.EXPIRY ||
            request.getType() == AdjustmentType.THEFT ||
            request.getType() == AdjustmentType.TRANSFER_OUT) {
            adjustmentAmount = -Math.abs(adjustmentAmount);
        }
        
        int newQty = previousQty + adjustmentAmount;
        
        if (newQty < 0) {
            throw new IllegalArgumentException("Insufficient stock. Current: " + previousQty);
        }
        
        medicine.setStockQuantity(newQty);
        medicineRepository.save(medicine);
        
        StockAdjustment adjustment = StockAdjustment.builder()
            .medicine(medicine)
            .previousQuantity(previousQty)
            .newQuantity(newQty)
            .adjustmentAmount(adjustmentAmount)
            .type(request.getType())
            .reason(request.getReason())
            .performedBy(username)
            .batchNumber(request.getBatchNumber())
            .build();
        
        StockAdjustment saved = adjustmentRepository.save(adjustment);
        
        log.info("Stock adjusted for medicine {}: {} -> {} ({})", 
            medicine.getName(), previousQty, newQty, request.getType());
        
        return toResponse(saved);
    }
    
    public List<StockAdjustmentResponse> getMedicineHistory(Long medicineId) {
        return adjustmentRepository.findByMedicineIdOrderByAdjustmentDateDesc(medicineId)
            .stream()
            .map(this::toResponse)
            .collect(Collectors.toList());
    }
    
    private StockAdjustmentResponse toResponse(StockAdjustment adjustment) {
        return StockAdjustmentResponse.builder()
            .id(adjustment.getId())
            .medicineId(adjustment.getMedicine().getId())
            .medicineName(adjustment.getMedicine().getName())
            .previousQuantity(adjustment.getPreviousQuantity())
            .newQuantity(adjustment.getNewQuantity())
            .adjustmentAmount(adjustment.getAdjustmentAmount())
            .type(adjustment.getType())
            .reason(adjustment.getReason())
            .performedBy(adjustment.getPerformedBy())
            .adjustmentDate(adjustment.getAdjustmentDate())
            .batchNumber(adjustment.getBatchNumber())
            .build();
    }
}
```
